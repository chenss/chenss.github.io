Matlabдёӯи®Ўз®—зЁӢеәҸиҝҗиЎҢж—¶й—ҙзҡ„дёүз§Қж–№жі•

жқҘжәҗпјҡhttp://www.matlabsky.com/thread-2607-1-1.html

з»ҸеёёжҲ‘д»¬йңҖиҰҒи®Ўз®—жҲ‘д»¬зЁӢеәҸеҲ°еә•иҝҗиЎҢеӨҡй•ҝж—¶й—ҙпјҢиҝҷж ·еҸҜд»ҘжҜ”иҫғзЁӢеәҸзҡ„жү§иЎҢж•ҲзҺҮгҖӮеҪ“з„¶иҝҷдёӘеҜ№дәҺеҸӘжңүеҮ з§’й’ҹзҡ„е°ҸзЁӢеәҸжІЎжңүд»Җд№Ҳж„Ҹд№үпјҢдҪҶжҳҜеҜ№дәҺеӨ§зЁӢеәҸе°ұжңүеҫҲйҮҚиҰҒзҡ„ж„Ҹд№үдәҶгҖӮ

дёӢйқўжҲ‘д»¬е°ұиҜҙиҜҙMatlabдёӯи®Ўз®—зЁӢеәҸиҝҗиЎҢж—¶й—ҙзҡ„дёүз§Қеёёз”Ёж–№жі•еҗ§пјҒ

жіЁж„Ҹпјҡдёүз§Қж–№жі•з”ұдәҺдҪҝз”ЁеҺҹзҗҶдёҚдёҖж ·пјҢеҫ—еҲ°з»“жһңеҸҜиғҪжңүдёҖе®ҡзҡ„е·®и·қпјҒ

1гҖҒticе’Ңtocз»„еҗҲ(жҺЁиҚҗ)

и®Ўз®—ticе’Ңtocд№Ӣй—ҙйӮЈж®өзЁӢеәҸд№Ӣй—ҙзҡ„иҝҗиЎҢж—¶й—ҙпјҢе®ғзҡ„з»Ҹе…ёж јејҸдёә

tic
гҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮ
toc

жҚўеҸҘиҜқиҜҙзЁӢеәҸпјҢзЁӢеәҸйҒҮеҲ°ticж—¶MatlabиҮӘеҠЁејҖе§Ӣи®Ўж—¶пјҢиҝҗиЎҢеҲ°tocж—¶иҮӘеҠЁи®Ўз®—жӯӨж—¶дёҺжңҖиҝ‘дёҖж¬Ўticд№Ӣй—ҙзҡ„ж—¶й—ҙгҖӮиҝҷдёӘжңүзӮ№жӢ—еҸЈпјҢдёӢйқўжҲ‘д»¬дёҫдёӘдҫӢеӯҗиҜҙжҳҺ

% by dynamic of MatlabжҠҖжңҜи®әеқӣ
% see also http://www.matlabsky.com
% contact me matlabsky@gmail.com
% 2009-08-18 12:08:47 
clc
tic;%tic1
t1=clock;
for i=1:3
    tic ;%tic2
    t2=clock;
    pause(3*rand)
    %и®Ўз®—еҲ°дёҠдёҖж¬ЎйҒҮеҲ°ticзҡ„ж—¶й—ҙпјҢжҚўеҸҘиҜқиҜҙе°ұжҳҜжҜҸж¬ЎеҫӘзҺҜзҡ„ж—¶й—ҙ
    disp(['tocи®Ўз®—з¬¬',num2str(i),'ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ',num2str(toc)]);
    %и®Ўз®—жҜҸж¬ЎеҫӘзҺҜзҡ„ж—¶й—ҙ
    disp(['etimeи®Ўз®—з¬¬',num2str(i),'ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ',num2str(etime(clock,t2))]);
    %и®Ўз®—зЁӢеәҸжҖ»е…ұзҡ„иҝҗиЎҢж—¶й—ҙ
    disp(['etimeи®Ўз®—зЁӢеәҸд»ҺејҖе§ӢеҲ°зҺ°еңЁиҝҗиЎҢзҡ„ж—¶й—ҙ:',num2str(etime(clock,t1))]);
    disp('======================================')
end
%и®Ўз®—жӯӨж—¶еҲ°tic2зҡ„ж—¶й—ҙпјҢз”ұдәҺжңҖеҗҺдёҖж¬ЎйҒҮеҲ°ticжҳҜеңЁforеҫӘзҺҜзҡ„i=3ж—¶пјҢжүҖд»Ҙи®Ўз®—зҡ„жҳҜжңҖеҗҺдёҖж¬ЎеҫӘзҺҜзҡ„ж—¶й—ҙ
disp(['tocи®Ўз®—жңҖеҗҺдёҖж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙ',num2str(toc)])
disp(['etimeзЁӢеәҸжҖ»иҝҗиЎҢж—¶й—ҙпјҡ',num2str(etime(clock,t1))]);

иҝҗиЎҢз»“жһңеҰӮдёӢпјҢеӨ§е®¶еҸҜд»ҘиҮӘе·ұеҲҶжһҗдёӢ

tocи®Ўз®—з¬¬1ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ2.5628
etimeи®Ўз®—з¬¬1ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ2.562
etimeи®Ўз®—зЁӢеәҸд»ҺејҖе§ӢеҲ°зҺ°еңЁиҝҗиЎҢзҡ„ж—¶й—ҙ:2.562
======================================
tocи®Ўз®—з¬¬2ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ2.8108
etimeи®Ўз®—з¬¬2ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ2.813
etimeи®Ўз®—зЁӢеәҸд»ҺејҖе§ӢеҲ°зҺ°еңЁиҝҗиЎҢзҡ„ж—¶й—ҙ:5.375
======================================
tocи®Ўз®—з¬¬3ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ2.0462
etimeи®Ўз®—з¬¬3ж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙпјҡ2.046
etimeи®Ўз®—зЁӢеәҸд»ҺејҖе§ӢеҲ°зҺ°еңЁиҝҗиЎҢзҡ„ж—¶й—ҙ:7.421
======================================
tocи®Ўз®—жңҖеҗҺдёҖж¬ЎеҫӘзҺҜиҝҗиЎҢж—¶й—ҙ2.0479
etimeзЁӢеәҸжҖ»иҝҗиЎҢж—¶й—ҙпјҡ7.421

2гҖҒetime(t1,t2)е№¶е’Ңclockй…ҚеҗҲ

жқҘи®Ўз®—t1пјҢt2д№Ӣй—ҙзҡ„ж—¶й—ҙе·®пјҢе®ғжҳҜйҖҡиҝҮи°ғз”Ёwindowsзі»з»ҹзҡ„ж—¶й’ҹиҝӣиЎҢж—¶й—ҙе·®и®Ўз®—еҫ—еҲ°иҝҗиЎҢж—¶й—ҙзҡ„пјҢеә”з”Ёзҡ„еҪўејҸ

t1=clock;
гҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮ
t2=clock;
etime(t2,t1)

иҮідәҺдҫӢеӯҗжҲ‘е°ұдёҚдёҫдәҶпјҢеӣ дёәеңЁдёҠйқўзҡ„дҫӢеӯҗдёӯдҪҝз”ЁдәҶetimeеҮҪж•°дәҶ

3гҖҒcputimeеҮҪж•°жқҘе®ҢжҲҗ

дҪҝз”Ёж–№жі•е’ҢetimeзӣёдјјпјҢеҸӘжҳҜиҝҷдёӘжҳҜдҪҝз”Ёcpuзҡ„дё»йў‘и®Ўз®—зҡ„пјҢе’ҢеүҚйқўеҺҹзҗҶдёҚеҗҢпјҢдҪҝз”Ёж јејҸеҰӮдёӢ

t0=cputime
гҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮгҖӮ
t1=cputime-t0

дёҠйқўиҜҙеҲ°дәҶдёүз§Қж–№жі•пјҢйғҪжҳҜеҸҜд»ҘиҝӣиЎҢзЁӢеәҸиҝҗиЎҢж—¶й—ҙи®Ўз®—зҡ„пјҢдҪҶжҳҜMatlabе®ҳж–№жҺЁиҚҗдҪҝз”Ёtic/tocз»„еҗҲпјҢWhen timing the duration of an event, use the tic and toc functions instead of clock or etime.

иҮідәҺеӨ§е®¶еҸҜд»Ҙж №жҚ®иҮӘе·ұзҡ„е–ңеҘҪиҮӘе·ұйҖүжӢ©пјҢдҪҶжҳҜдҪҝз”Ёtic/tocзҡ„ж—¶еҖҷдёҖе®ҡиҰҒжіЁж„ҸпјҢtocи®Ўз®—зҡ„жҳҜдёҺжңҖеҗҺдёҖж¬ЎиҝҗиЎҢзҡ„ticд№Ӣй—ҙзҡ„ж—¶й—ҙпјҢдёҚжҳҜз¬¬дёҖдёӘticпјҢжӣҙдёҚжҳҜз¬¬дәҢдёӘгҖӮгҖӮгҖӮгҖӮгҖӮ
