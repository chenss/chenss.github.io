з»ҸиҝҮиҝҷз•Әдҝ®ж”№пјҢеҸҜд»Ҙи§ЈеҶіиЎЁеҚ•ж–№ејҸзҡ„POSTжҸҗдәӨHTTP 403й—®йўҳдәҶгҖӮajaxзҡ„PostжҸҗдәӨд»…иҝҷж ·ж”№иҝҳдёҚиЎҢгҖӮиҝҳйңҖиҰҒй’©дёҠдёҖдёӘжҜҸж¬ЎжҸҗдәӨж—¶зҡ„cookieеӨ„зҗҶиҝҮзЁӢгҖӮд№ҹе°ұжҳҜжҜҸж¬ЎжҸҗдәӨзҡ„ж—¶еҖҷпјҢйғҪи§ҰеҸ‘иҝҷдёӘиҝҮзЁӢпјҢеңЁжҸҗдәӨзҡ„httpеӨҙдёҠеҠ е…Ҙcsrf tokenгҖӮдёҚиҝҮиҝҳеҘҪпјҢеҰӮжһңдҪ жҳҜз”ЁjQueryжқҘеӨ„зҗҶajaxзҡ„иҜқпјҢDjangoзӣҙжҺҘйҖҒдәҶдёҖж®өи§ЈеҶій—®йўҳзҡ„д»Јз ҒгҖӮжҠҠе®ғж”ҫеңЁдёҖдёӘзӢ¬з«Ӣзҡ„jsж–Үд»¶дёӯпјҢеңЁhtmlйЎөйқўдёӯйғҪеј•е…ҘеҚіеҸҜгҖӮжіЁж„ҸиҝҷдёӘjsж–Үд»¶еҝ…йЎ»еңЁjqueryзҡ„jsж–Үд»¶еј•е…Ҙд№ӢеҗҺпјҢеҶҚеј•е…ҘгҖӮд»Јз ҒжҲ‘зӣҙжҺҘеӨҚеҲ¶иҝҮжқҘдәҶпјҢеҰӮдёӢпјҡ

$('html').ajaxSend(function(event, xhr, settings) {
    function getCookie(name) {
        var cookieValue = null;
        if (document.cookie && document.cookie != '') {
            var cookies = document.cookie.split(';');
            for (var i = 0; i < cookies.length; i++) {
                var cookie = jQuery.trim(cookies[i]);
                // Does this cookie string begin with the name we want?
                if (cookie.substring(0, name.length + 1) == (name + '=')) {
                    cookieValue = decodeURIComponent(cookie.substring(name.length + 1));
                    break;
                }
            }
        }
        return cookieValue;
    }
    if (!(/^http:.*/.test(settings.url) || /^https:.*/.test(settings.url))) {
        // Only send the token to relative URLs i.e. locally.
        xhr.setRequestHeader("X-CSRFToken", getCookie('csrftoken'));
    }
});
